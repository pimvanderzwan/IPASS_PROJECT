501 class glcd_oled_i2c_128x64_fast_buffered : public ssd1306_i2c, public window {
502 private:
503 
504  static auto constexpr wsize = xy( 128, 64 );
505 
506  static auto constexpr buf_size = wsize.x * wsize.y / 8;
507 
508  uint8_t buffer[ buf_size ];
509  bool dirty[ buf_size ];
510 
511  void write_implementation(
512  xy pos,
513  color col
514  ) override {
515  int a = pos.x + ( pos.y / 8 ) * size.x;
516 
517  if( col == white ){
518  buffer[ a ] |= ( 0x01 << (pos.y % 8 ));
519  } else {
520  buffer[ a ] &= ~( 0x01 << ( pos.y % 8 ));
521  }
522 
523  dirty[ a ] = true;
524  }
525 
526 public:
527 
529  glcd_oled_i2c_128x64_fast_buffered( i2c_bus & bus, int address = 0x3C ):
530  ssd1306_i2c( bus, address ),
531  window( wsize, white, black )
532  {
533  bus.write( address ).write(
534  ssd1306_initialization,
535  sizeof( ssd1306_initialization ) / sizeof( uint8_t )
536  );
537  }
538 
539  void flush() override {
540  unsigned int start = 0;
541  while( start < buf_size ){
542 
543  // find first/next dirty pixel
544  while( !dirty[ start ] ){
545  ++start;
546  if( start == buf_size ){
547  // no dirty pixels: nothing to do
548  return;
549  }
550  }
551 
552  // find the extent of the next message
553  unsigned int tail = start;
554  unsigned int cursor = start;
555  unsigned int gap = 0;
556  for(;;){
557  if( dirty[ cursor ] ){
558  gap = 0;
559  tail = cursor;
560  dirty[ cursor ] = false;
561  } else {
562  // if gap larger than 10: flush upto last visited dirty pixel
563  if( ++gap > 10 ){
564  break;
565  }
566  }
567  ++cursor;
568  // if end of buffer: flush
569  if( cursor == buf_size ){
570  break;
571  }
572  }
573 
574  command( ssd1306_commands::column_addr, start % wsize.x, 127 );
575  command( ssd1306_commands::page_addr, start / wsize.x, 7 );
576  auto t = bus.write( address );
577  t.write( ssd1306_data_prefix );
578  t.write( & buffer[ start ], 1 + ( tail - start ) );
579  }
580  }
581 
582 }; // class glcd_oled_i2c_128x64_fast_buffered
